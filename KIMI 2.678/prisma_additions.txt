model RawPost {
  id              String   @id @default(cuid())
  platform        String
  nativeId        String
  entityType      String   // company | employee | product
  entityId        String
  content         String   @db.Text
  contentType     String
  url             String?
  mediaUrl        String?
  postedAt        DateTime
  collectedAt     DateTime @default(now())
  rawReach        Int?
  rawLikes        Int?
  rawReplies      Int?
  rawReposts      Int?
  rawClicks       Int?
  rawOpens        Int?
  rawComments     Int?
  rawStars        Int?
  rawForks        Int?
  rawMetrics      Json?
  extractedUrls   String[]
  rawHash         String   @unique
  normalizedToPostId String? @unique
  @@index([platform, entityId, postedAt])
  @@index([rawHash])
}

model Company {
  id              String   @id @default(cuid())
  name            String
  slug            String   @unique
  description     String?
  youtubeSubscribers Int?
  youtubeViews    Int?
  youtubeVideos   Int?
  newsletterSubscribers Int?
  xFollowers      Int?
  linkedinFollowers Int?
  githubRepos     Int?
  productHuntProducts Int?
  podcastEpisodes Int?
  teamSize        Int?
  publicFacingCount Int?
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
}

model TrackedRedirect {
  id              String   @id @default(cuid())
  shortId         String   @unique
  postId          String?
  destinationUrl  String
  clickCount      Int      @default(0)
  utmCampaign     String?
  utmMedium       String?
  utmContent      String?
  createdAt       DateTime @default(now())
  @@index([shortId])
}

model ConversionEvent {
  id              String   @id @default(cuid())
  redirectId      String?
  postId          String?
  entityType      String
  entityId        String
  eventType       String
  referrerUrl     String?
  utmCampaign     String?
  utmMedium       String?
  utmContent      String?
  userAgent       String?
  ipAddress       String?
  occurredAt      DateTime @default(now())
  @@index([entityType, entityId, eventType])
  @@index([occurredAt])
}
