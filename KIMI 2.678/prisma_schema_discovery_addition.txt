
// === EXHAUSTIVE SURFACE DISCOVERY ===
// Added for Phase 4: Signal Capture

model DiscoveredSurface {
  id              String   @id @default(cuid())
  employeeId      String
  employee        Employee @relation(fields: [employeeId], references: [id])

  surface         String   // platform name: x, linkedin, github, substack, etc.
  label           String   // human-readable label
  handle          String?  // username/handle
  url             String?  // direct URL
  status          String   @default("unknown") // unknown | verified | likely | absent | claimed

  // Discovery metadata
  discoveryMethod String   // manual | search_api | spider | x_api | github_api | linkedin_scrape | rss | sitemap
  discoveredAt    DateTime @default(now())
  lastVerifiedAt  DateTime?
  confidenceScore Float    @default(0) // 0-1

  // Raw evidence
  evidence        String[] // URLs, screenshots, search results that confirm this surface
  notes           String?

  // For search-derived surfaces
  searchQuery     String?  // the query that found this
  searchRank      Int?     // position in search results

  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt

  @@unique([employeeId, surface, handle])
  @@index([employeeId])
  @@index([surface])
  @@index([status])
}

model DiscoveryJob {
  id            String   @id @default(cuid())
  jobType       String   // full_roster | employee | surface_type
  status        String   @default("pending") // pending | running | completed | failed

  // Scope
  employeeId    String?  // null = full roster
  surface       String?  // null = all surfaces

  // Results
  surfacesFound Int      @default(0)
  surfacesUpdated Int    @default(0)
  errors        String[]

  startedAt     DateTime?
  completedAt   DateTime?

  createdAt     DateTime @default(now())

  @@index([status])
  @@index([employeeId])
}
